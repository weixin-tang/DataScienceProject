def getNullPercent( df  ):
    df         =    ( df.isnull().sum()/len( df )*100)
    df         =    pd.DataFrame( df.sort_values( ascending=False ) ) 
    df.columns =    ["Percent"]     
    df         =    df[ df["Percent"] != 0  ]
    return df

def getFeatureAndTarget( df ):
    features = df.iloc[:,:-1]
    targets = df.iloc[:,-1]
    return features , targets
    
def setColToEnd( df , endCol  ):
    myList =  list( (set( df.columns )).difference( { endCol } ) ) #set( list(endCol) ) {'SalePrice'}
    #print( myList )
    myList.append( endCol )
    return  df[myList]

def printTrainTestScore( model , X_train, X_test, y_train, y_test   ):
    print('Accuracy of classifier on training set: {:.2f}'
         .format(model.score( X_train ,  y_train )))   
    print('Accuracy of classifier on test set: {:.2f}'
         .format(model.score(X_test , y_test )))
    
def trainTestSplit(  features , target   ):
    from sklearn.model_selection import train_test_split
    X_train, X_test, y_train, y_test = train_test_split( features , target , test_size=0.3, random_state=2)
    return X_train, X_test, y_train, y_test

def regressionModelEvaluation( y_test, y_pred ):
    print('Mean Absolute Error:', metrics.mean_absolute_error(y_test, y_pred))  
    print('Mean Squared Error:', metrics.mean_squared_error(y_test, y_pred))  
    print('Root Mean Squared Error:', np.sqrt(metrics.mean_squared_error(y_test, y_pred)))
    print('R-squared Error:', metrics.r2_score(y_test, y_pred))

# so we need remove outlier 
def getBoxPlot( df , colName ):
    # easy way to see outlier
    import seaborn as sns
    sns.boxplot(x=df[colName])
    plt.show()

def getLogTransform( df , colName , newColName  ):
    df[newColName] = (df[colName]+1).transform(np.log)
    return df

def getRemoveOutlier( df , colName ):
    Q1 = df[colName].quantile(0.25)
    Q3 = df[colName].quantile(0.75)
    IQR = Q3 - Q1
    return df[(df[colName] < (Q3 + 1.5 * IQR)) & (df[colName] > (Q1 - 1.5 * IQR))]

def getModelByTrainWithRandomForestInsideBagging( X_train, y_train ):
    from sklearn.ensemble import RandomForestClassifier, BaggingClassifier
    rf = RandomForestClassifier(n_estimators=10)  
    bag_clf = BaggingClassifier(
                            base_estimator=rf, # the model inside
                            n_estimators=10,   # how many small part to train
                            bootstrap=True,    # whether sample will replacement
                            n_jobs=-1,         # means make things paralle
                            random_state=42)

    bag_clf.fit(X_train, y_train)
    return bag_clf
